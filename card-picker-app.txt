function onFile(file){
  if(!file)return;
  var reader=new FileReader();
  reader.onload=function(e){
    var rows=csvParse(e.target.result);
    if(!rows.length){showMsg("No data found","var(--red)");return}
    var cols=Object.keys(rows[0]);
    var isPerk=cols.some(function(c){return["perk","credit","benefit"].indexOf(c)>=0});
    if(isPerk){
      D.perks=toPerks(rows);save();
      showMsg("✅ Loaded "+D.perks.length+" perks!","var(--grn)");
    }else{
      D.cards=toCards(rows);save();
      showMsg("✅ Loaded "+D.cards.length+" cards!","var(--grn)");
    }
    setTimeout(function(){view="search";render()},1200);
  };
  reader.readAsText(file);
}

function detectLocation(){
  if(!navigator.geolocation){locState="error";render();return}
  locState="loading";nearbyPlaces=[];render();
  navigator.geolocation.getCurrentPosition(function(pos){
    var lat=pos.coords.latitude,lon=pos.coords.longitude;
    // Use Overpass API to find nearby shops/restaurants/amenities
    var radius=100; // meters
    var overpassQuery='[out:json][timeout:10];(node["name"](around:'+radius+','+lat+','+lon+');way["name"]["shop"](around:'+radius+','+lat+','+lon+');way["name"]["amenity"](around:'+radius+','+lat+','+lon+'););out body 10;';
    fetch("https://overpass-api.de/api/interpreter",{method:"POST",body:"data="+encodeURIComponent(overpassQuery)})
    .then(function(r){return r.json()})
    .then(function(data){
      var places=[];
      var seen={};
      if(data.elements){
        data.elements.forEach(function(el){
          var name=el.tags&&el.tags.name;
          if(!name||seen[name.toLowerCase()])return;
          seen[name.toLowerCase()]=true;
          var type=el.tags.shop||el.tags.amenity||el.tags.cuisine||el.tags.leisure||"";
          places.push({name:name,type:type});
        });
      }
      if(places.length===0){
        // Fallback: try wider radius
        var overpassQuery2='[out:json][timeout:10];(node["name"]["shop"](around:300,'+lat+','+lon+');node["name"]["amenity"~"restaurant|cafe|fast_food|fuel|pharmacy|bank"](around:300,'+lat+','+lon+');way["name"]["shop"](around:300,'+lat+','+lon+'););out body 15;';
        return fetch("https://overpass-api.de/api/interpreter",{method:"POST",body:"data="+encodeURIComponent(overpassQuery2)}).then(function(r){return r.json()}).then(function(data2){
          if(data2.elements){
            data2.elements.forEach(function(el){
              var name=el.tags&&el.tags.name;
              if(!name||seen[name.toLowerCase()])return;
              seen[name.toLowerCase()]=true;
              var type=el.tags.shop||el.tags.amenity||el.tags.cuisine||"";
              places.push({name:name,type:type});
            });
          }
          nearbyPlaces=places.slice(0,8);
          locState=nearbyPlaces.length?"done":"error";
          if(nearbyPlaces.length===1){query=nearbyPlaces[0].name;locName=nearbyPlaces[0].name}
          render();
        });
      }
      nearbyPlaces=places.slice(0,8);
      locState=nearbyPlaces.length?"done":"error";
      if(nearbyPlaces.length===1){query=nearbyPlaces[0].name;locName=nearbyPlaces[0].name}
      render();
    })
    .catch(function(e){
      console.error("Overpass error",e);
      // Fallback to Nominatim reverse geocode
      fetch("https://nominatim.openstreetmap.org/reverse?lat="+lat+"&lon="+lon+"&format=json&zoom=18&addressdetails=1")
      .then(function(r){return r.json()})
      .then(function(data){
        if(data&&data.name){
          nearbyPlaces=[{name:data.name,type:""}];
          locName=data.name;
          query=data.name;
          locState="done";
        }else if(data&&data.address){
          var addr=data.address;
          var placeName=addr.shop||addr.amenity||addr.building||addr.road||"";
          if(placeName){nearbyPlaces=[{name:placeName,type:""}];locName=placeName;query=placeName;locState="done"}
          else{locState="error"}
        }else{locState="error"}
        render();
      }).catch(function(){locState="error";render()});
    });
  },function(err){
    console.error("Geo error",err);
    locState="error";render();
  },{enableHighAccuracy:true,timeout:10000,maximumAge:60000});
}
